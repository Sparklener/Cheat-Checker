    for(int j = 0;j <= m;j ++)
    for(int k = 0;k <= m - j;k ++)
    {
        f[i][j][k] = f[i - 1][j][k];
        if(k > 0) f[i][j][k] += f[i - 1][j + 1][k - 1] * (j + 1);

        if(k > 1) f[i][j][k] += f[i - 1][j + 2][k - 2] * c[j + 2];
        f[i][j][k] %= MOD;
    }
    ll ans = 0;
    for(int i = 0;i <= m;i ++)
        for(int j = 0;j <= m;j ++)
